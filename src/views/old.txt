<template>
  <v-container
    fluid
    class="login-container fill-height d-flex align-center justify-center"
  >
    <div class="login-overlay"></div>
    <v-row dense class="mt-32 mt-md-0">
      <v-col cols="12">
        <v-container fluid class="d-flex align-center justify-center">
          <v-card class="login-card" elevation="24">
            <v-card-text>
              <div class="text-center mb-0 mb-md-6">
                <v-img
                  :src="logo"
                  contain
                  max-height="100"
                  class="max-height-md-400 max-height-lg-600"
                />
              </div>

              <v-form @submit.prevent="handleSubmit">
                <v-text-field
                  v-model="email"
                  label="Email"
                  variant="outlined"
                  color="success"
                  prepend-inner-icon="mdi-email"
                  :rules="emailRules"
                  class="mt-1 mb-4 green-input"
                />
                <v-text-field
                  v-model="password"
                  :type="showPassword ? 'text' : 'password'"
                  label="Contraseña"
                  variant="outlined"
                  color="success"
                  prepend-inner-icon="mdi-lock"
                  :append-inner-icon="showPassword ? 'mdi-eye' : 'mdi-eye-off'"
                  @click:append-inner="showPassword = !showPassword"
                  :rules="passwordRules"
                  class="mb-4 green-input"
                />
                <v-btn
                  block
                  color="#00ff88"
                  type="submit"
                  :loading="loading"
                  class="font-orbitron"
                >
                  <v-icon start>mdi-login</v-icon>
                  Iniciar sesión
                </v-btn>
                <v-row>
                  <v-col class="pl-14 pr-14">
                    <v-btn
                      block
                      variant="outlined"
                      color="#00ff88"
                      to="/register"
                      class="mt-4"
                      style="font-size: 10px"
                    >
                      ¿No tienes cuenta? Regístrate
                    </v-btn>
                  </v-col>
                </v-row>
                <v-alert v-if="error" type="error" class="mt-4">
                  {{ error }}
                </v-alert>
              </v-form>
            </v-card-text>
          </v-card>
        </v-container>
      </v-col>
    </v-row>
  </v-container>
</template>

<script setup>
import { ref } from "vue";
import { useAuthStore } from "@/stores/auth.store";
import { useRouter } from "vue-router";
import logo from "@/assets/logo.svg";

const email = ref("");
const password = ref("");
const showPassword = ref(false);
const loading = ref(false);
const error = ref("");

const emailRules = [
  (v) => !!v || "Email es requerido",
  (v) => /.+@.+\..+/.test(v) || "Email debe ser válido",
];

const passwordRules = [
  (v) => !!v || "Contraseña es requerida",
  (v) => (v && v.length >= 8) || "Contraseña debe tener al menos 8 caracteres",
];

const authStore = useAuthStore();
const router = useRouter();

const handleSubmit = async () => {
  try {
    loading.value = true;
    error.value = "";

    await authStore.login(email.value, password.value);

    if (authStore.user.isVerified) {
      router.push("/dashboard");
    } else {
      router.push("/verify");
    }
  } catch (err) {
    error.value = err.response?.data?.message || "Error al iniciar sesión";
  } finally {
    loading.value = false;
  }
};
</script>
